```bash
# Install
$ npm install -g extra-bel.sh
```

```bash
$ ebel [<count>]

# [] -> optional argument
# <> -> argument value
```

<br>

```bash
# Produce one BEL
$ ebel

# Produce 4 BELs
$ ebel 4

# Produce 4 BELs after copy is over
$ cp src.zip dst.zip && ebel 4
```
